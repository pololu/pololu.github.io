51  """
52  Sends the "Get firmware version" command to get the device's firmware

70  def set_protocol_options(self, options):
71  """
72  Sends the "Set protocol options" command to the device to specify options
73  related to how the device processes commands and sends responses.
74  The options are also saved in this object and are used later
75  when sending other commands or reading responses.
76 
77  When CRC for commands is enabled, this library generates the CRC
78  byte and appends it to the end of each command it sends. The Motoron
79  checks it to help ensure the command was received correctly.
80 
81  When CRC for responses is enabled, this library reads the CRC byte sent
82  by the Motoron in its repsonses and makes sure it is correct. If the
83  response CRC byte is incorrect, get_last_error() will return a non-zero
84  error code after the command has been run.
85 
86  When the I2C general call address is enabled, the Motoron receives
87  commands sent to address 0 in addition to its usual I2C address.
88  The general call address is write-only; reading bytes from it is not
89  supported.
90 
91  By default (self, in this libary and the Motoron itself), CRC for commands and
92  responses is enabled, and the I2C general call address is enabled.
93 
94  This method always sends its command with a CRC byte, so it will work
95  even if CRC was previously disabled but has been re-enabled on the device
96  (e.g. due to a reset).
97 
98  The \p options argument should be 0 or combination of the following
99  expressions made using the bitwise or operator (|):
100  - (1 << motoron.PROTOCOL_OPTION_CRC_FOR_COMMANDS)
101  - (1 << motoron.PROTOCOL_OPTION_CRC_FOR_RESPONSES)
102  - (1 << motoron.PROTOCOL_OPTION_I2C_GENERAL_CALL)
103 
104  For more information, see the "Set protocol optons"
105  command in the Motoron user's guide.
106 
107  \sa enable_crc(), disable_crc(),
108  enable_crc_for_commands(), disable_crc_for_commands(),
109  enable_crc_for_responses(), disable_crc_for_responses(),
110  enable_i2c_general_call(), disable_i2c_general_call()
111  """
112  self.__protocol_options = options
113  cmd = [
114  CMD_SET_PROTOCOL_OPTIONS,
115  options & 0x7F,
116  ~options & 0x7F
117  ]
118  self.__send_command_core(cmd, True)

213  def write_eeprom(self, offset, value):
214  """
215  Writes a value to one byte in the Motoron's EEPROM memory.
216 
217  **Warning: Be careful not to write to the EEPROM in a fast loop. The
218  EEPROM memory of the Motoron's microcontroller is only rated for
219  100,000 erase/write cycles.**
220 
221  For more information, see the "Write EEPROM" command in the
222  Motoron user's guide.
223  """
224  cmd = [
225  CMD_WRITE_EEPROM,
226  offset & 0x7F,
227  value & 0x7F,
228  (value >> 7) & 1,
229  ]
230  cmd += [
231  cmd[1] ^ 0x7F,
232  cmd[2] ^ 0x7F,
233  cmd[3] ^ 0x7F,
234  ]
235  self.__send_command(cmd)
236  time.sleep(0.006)

267  def reset(self, ignore_nack=True):
268  """
269  Sends a "Reset" command to the Motoron, which does a full hardware reset.
270 
271  This command is equivalent to briefly driving the Motoron's RST pin low.
272  The Motoron's RST is briefly driven low by the Motoron itself as a
273  result of this command.
274 
275  After running this command, we recommend waiting for at least 5
276  milliseconds before you try to communicate with the Motoron.
277 
278  \param ignore_nack Optional argument: if `True` (the default), this method
279  ignores a NACK error if it occurs on sending the Reset command. This is
280  useful in case the Motoron has CRC off and executes the reset before it
281  can ACK the CRC byte (which this method always sends to make it more
282  reliable).
283 
284  \sa reinitialize()
285  """
286  # Always send the reset command with a CRC byte to make it more reliable.
287  cmd = [CMD_RESET]
288  try:
289  self.__send_command_core(cmd, True)
290  except OSError as e:
291  # Errno 5 (Input/output error) or 121 (Remote I/O error) indicates a
292  # NACK of a data byte. Ignore it if the ignore_nack argument is True.
293  # In all other cases, re-raise the exception.
294  if not (ignore_nack and (e.args[0] == 5 or e.args[0] == 121)): raise
295  self.__protocol_options = MotoronI2C.__DEFAULT_PROTOCOL_OPTIONS

353  def get_status_flags(self):
354  """
355  Reads the "Status flags" variable from the Motoron.
356 
357  The bits in this variable are defined by the STATUS_FLAGS_*
358  constants in the motoron package:
359 
360  - motoron.STATUS_FLAG_PROTOCOL_ERROR
361  - motoron.STATUS_FLAG_CRC_ERROR
362  - motoron.STATUS_FLAG_COMMAND_TIMEOUT_LATCHED
363  - motoron.STATUS_FLAG_MOTOR_FAULT_LATCHED
364  - motoron.STATUS_FLAG_NO_POWER_LATCHED
365  - motoron.STATUS_FLAG_RESET
366  - motoron.STATUS_FLAG_COMMAND_TIMEOUT
367  - motoron.STATUS_FLAG_MOTOR_FAULTING
368  - motoron.STATUS_FLAG_NO_POWER
369  - motoron.STATUS_FLAG_ERROR_ACTIVE
370  - motoron.STATUS_FLAG_MOTOR_OUTPUT_ENABLED
371  - motoron.STATUS_FLAG_MOTOR_DRIVING
372 
373  Here is some example code that uses bitwise operators to check
374  whether there is currently a motor fault or a lack of power:
375 
376  ```{.py}
377  mask = ((1 << motoron.STATUS_FLAG_NO_POWER) |
378  (1 << motoron.STATUS_FLAG_MOTOR_FAULTING))
379  if mc.get_status_flags() & mask: # do something
380  ```
381 
382  This library has helper methods that make it easier if you just want to
383  read a single bit:
384 
385  - get_protocol_error_flag()
386  - get_crc_error_flag()
387  - get_command_timeout_latched_flag()
388  - get_motor_fault_latched_flag()
389  - get_no_power_latched_flag()
390  - get_reset_flag()
391  - get_motor_faulting_flag()
392  - get_no_power_flag()
393  - get_error_active_flag()
394  - get_motor_output_enabled_flag()
395  - get_motor_driving_flag()
396 
397  The clear_latched_status_flags() method sets the specified set of latched
398  status flags to 0. The reinitialize() and reset() commands reset the
399  latched status flags to their default values.
400 
401  For more information, see the "Status flags" variable in the Motoron
402  user's guide.
403  """
404  return self.get_var_u16(0, VAR_STATUS_FLAGS)

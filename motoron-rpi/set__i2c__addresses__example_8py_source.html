9help_message = """
10To assign an I2C address to a Motoron, type "a" followed by the address
11(in decimal) while the JMP1 pin of the Motoron you wish to change is
12shorted to GND. For example, "a7" sets the address of the Motoron to
13decimal 17.
14
15Alternatively, you can type "a" by itself in order to automatically
16to have this program automatically pick an address for you.
17
18To make the Motoron start using its new address, you can reset it by
19power cycling it, driving its RST line low, or by typing "r".
20
21The "s" command does a simple scan of the bus to see which addresses
22have devices responding. This can help you make sure you have set up
23your I2C bus correctly.
24
25The "i" command goes further, sending extra commands to identify the
26Motoron devices on the bus and print information about them.
27
28Warning: The "i" command sends Motoron commands to every I2C device on
29the bus. If you have devices on your bus that are not Motorons, this
30could cause undesired behavior.
31
32Warning: The "a" and "r" commands use the I2C general call address (0), so
33they might cause undesired behavior on other devices that use that address.
34Also, they will not work if you disabled the general call address on a
35Motoron.
36"""

52# Make a MotoronI2C object configured to use the general call address (0).
53mc = motoron.MotoronI2C(address=0)

55# The next address this program will try to use when you send an "a" command
56# to automatically assign an address.
57next_address = 17

59# This function defines which I2C addresses this program is
60# allowed to communicate with.
61def allow_address_communication(address):
62 # Addresses cannot be larger than 127.
63 if address >= 128: return False

65 # If you have devices on your bus and you want to prevent this
66 # sketch from talking to them, potentially causing unwanted

70 return True

72# This function defines which I2C addresses this program is allowed
73# to communicate with.
74def allow_address_assignment(address):
75 return allow_address_communication(address) and address != 0
76
77# Checks to see if a device is responding at the specified address.
78def scan_address(address):
79 if not allow_address_communication(address): return False
80 try:
81 mc.bus.i2c_rdwr(i2c_msg.write(address, []))
82 return True
83 except OSError as e:
84 if e.args[0] == 6 or e.args[0] == 121: return False
85 raise

87def assign_address(line):
88 global next_address
89 desired_address_specified = False
90 desired_address = next_address
91 if len(line) > 1:
92 try:
93 desired_address = int(line[1:]) & 127
94 desired_address_specified = True
95 except ValueError:
96 pass
97
98 while True:
99 if allow_address_assignment(desired_address):
100 # Unless the address was explicitly specified by the user,
101 # make sure it is not already used by some other device.
102 if desired_address_specified or not scan_address(desired_address):
103 break
104 print("Found a device at address %d." % desired_address)
105
106 elif desired_address_specified:
107 print("Assignment to address %d not allowed." % desired_address)
108
109 # Try the next higher address.
110 while True:
111 desired_address = (desired_address + 1) & 127
112 if allow_address_assignment(desired_address): break
113
114 mc.enable_crc()
115
116 # Send a command to set the EEPROM device number to the desired
117 # address. This command will only be received by Motorons that
118 # are configured to listen to the general call address (which is
119 # the default) and will only be obeyed by Motorons that have JMP1
120 # shorted to GND.
121 mc.write_eeprom_device_number(desired_address)
122
123 print("Assigned address", desired_address)
124 next_address = desired_address + 1

126def scan_for_devices():
127 print("Scanning for I2C devices...")
128 for i in range(0, 128):
129 if scan_address(i):
130 print("Found device at address", i)
131
132# Note: This routine writes to many different addresses on your
133# I2C bus. If you have any non-Motoron devices on your bus,
134# this could cause unexpected changes to those devices.
135# You can modify allow_address_communication() to prevent this.
136def identify_devices():
137 print("Identifying Motoron controllers...")
138 for i in range(1, 128):
139 if not allow_address_communication(i): continue
140 test = motoron.MotoronI2C(bus=mc.bus, address=i)
141 try:
142 # Multiple Motorons on the same address sending different responses will
143 # cause CRC errors but we would like to ignore them.
144 test.disable_crc_for_responses()
145 v = test.get_firmware_version()
146 jumper_state = test.get_jumper_state() & 3
147 except OSError:
148 continue
149 jumper_string = ['both', 'on', 'off', 'err'][jumper_state]
150 print("%3d: product=0x%04X version=%x.%02x JMP1=%s" % (i, v['product_id'],
151 v['firmware_version']['major'], v['firmware_version']['minor'], jumper_string))

153def process_input_line(line):
154 if not line: return
155 elif line[0] == 'a': assign_address(line)

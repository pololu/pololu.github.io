61 def set_protocol_options(self, options):
62 """
63 Sends the "Set protocol options" command to the device to specify options
64 related to how the device processes commands and sends responses.
65 The options are also saved in this object and are used later
66 when sending other commands or reading responses.
67
68 When CRC for commands is enabled, this library generates the CRC
69 byte and appends it to the end of each command it sends. The Motoron
70 checks it to help ensure the command was received correctly.
71
72 When CRC for responses is enabled, this library reads the CRC byte sent
73 by the Motoron in its repsonses and makes sure it is correct. If the
74 response CRC byte is incorrect, get_last_error() will return a non-zero
75 error code after the command has been run.
76
77 When the I2C general call address is enabled, the Motoron receives
78 commands sent to address 0 in addition to its usual I2C address.
79 The general call address is write-only; reading bytes from it is not
80 supported.
81
82 By default (in this libary and the Motoron itself), CRC for commands and
83 responses is enabled, and the I2C general call address is enabled.
84
85 This method always sends its command with a CRC byte, so it will work
86 even if CRC was previously disabled but has been re-enabled on the device
87 (e.g. due to a reset).
88
89 The \p options argument should be 0 or combination of the following
90 expressions made using the bitwise or operator (|):
91 - (1 << motoron.PROTOCOL_OPTION_CRC_FOR_COMMANDS)
92 - (1 << motoron.PROTOCOL_OPTION_CRC_FOR_RESPONSES)
93 - (1 << motoron.PROTOCOL_OPTION_I2C_GENERAL_CALL)
94
95 For more information, see the "Set protocol optons"
96 command in the Motoron user's guide.
97
98 \sa enable_crc(), disable_crc(),
99 enable_crc_for_commands(), disable_crc_for_commands(),
100 enable_crc_for_responses(), disable_crc_for_responses(),
101 enable_i2c_general_call(), disable_i2c_general_call()
102 """
103 self.protocol_options = options
104 cmd = [
105 CMD_SET_PROTOCOL_OPTIONS,
106 options & 0x7F,
107 ~options & 0x7F
108 ]
109 self._send_command_core(cmd, True)

350 def reset(self, ignore_nack=True):
351 """
352 Sends a "Reset" command to the Motoron, which does a full hardware reset.
353
354 This command is equivalent to briefly driving the Motoron's RST pin low.
355 The Motoron's RST is briefly driven low by the Motoron itself as a
356 result of this command.
357
358 After running this command, we recommend waiting for at least 5
359 milliseconds before you try to communicate with the Motoron.
360
361 \param ignore_nack Optional argument: if `True` (the default), this method
362 ignores a NACK error if it occurs on sending the Reset command. This is
363 useful in case the Motoron has CRC off and executes the reset before it
364 can ACK the CRC byte (which this method always sends to make it more
365 reliable).
366
367 \sa reinitialize()
368 """
369 # Always send the reset command with a CRC byte to make it more reliable.
370 cmd = [CMD_RESET]
371 try:
372 self._send_command_core(cmd, True)
373 except OSError as e:
374 # Errno 5 (Input/output error) or 121 (Remote I/O error) indicates a
375 # NACK of a data byte. Ignore it if the ignore_nack argument is True.
376 # In all other cases, re-raise the exception.
377 if not (ignore_nack and (e.args[0] == 5 or e.args[0] == 121)): raise
378 self.protocol_options = MotoronBase.__DEFAULT_PROTOCOL_OPTIONS

1613def calculate_current_limit(milliamps, type, reference_mv, offset):
1614 """
1615 Calculates a current limit value that can be passed to the Motoron
1616 using set_current_limit().
1617
1618 \param milliamps The desired current limit, in units of mA.
1619 \param type Specifies what type of Motoron you are using. This should be one
1620 of the members of the motoron.CurrentSenseType enum.

1622 For example, use 3300 for a 3.3 V system or 5000 for a 5 V system.

1625 Motoron's "Current sense offset" variable using set_current_sense_offset(),

1628 (50*1024/reference_mv) but it can vary widely.

1630 if milliamps > 1000000: milliamps = 1000000
1631 limit = offset * 125 / 128 + milliamps * 20 / (reference_mv * (type.value & 3))
1632 if limit > 1000: limit = 1000
1633 return math.floor(limit)

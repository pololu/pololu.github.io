1328 static uint8_t calculateCrc(uint8_t length, const uint8_t * buffer)
1329 {
1330 uint8_t crc = 0;
1331 for (uint8_t i = 0; i < length; i++)
1332 {
1333 crc = pgm_read_byte(&motoronCrcTable[crc ^ buffer[i]]);
1334
1335 // The code below shows an alternative method to calculate the CRC
1336 // without using a lookup table. It will generally be slower but save
1337 // program space.
1338 // crc ^= buffer[i];
1339 // for (uint8_t j = 0; j < 8; j++)
1340 // {
1341 // if (crc & 1) { crc ^= 0x91; }
1342 // crc >>= 1;
1343 // }
1344 }
1345 return crc;
1346 }

1350 void sendCommandCore(uint8_t length, const uint8_t * cmd, bool sendCrc)
1351 {
1352 bus->beginTransmission(address);
1353 for (uint8_t i = 0; i < length; i++)
1354 {
1355 bus->write(cmd[i]);
1356 }
1357 if (sendCrc)
1358 {
1359 bus->write(calculateCrc(length, cmd));
1360 }
1361 lastError = bus->endTransmission();
1362 }
1363
1364 void sendCommand(uint8_t length, const uint8_t * cmd)
1365 {
1366 bool sendCrc = protocolOptions & (1 << MOTORON_PROTOCOL_OPTION_CRC_FOR_COMMANDS);
1367 sendCommandCore(length, cmd, sendCrc);
1368 }

1370 void readResponse(uint8_t length, uint8_t * response)
1371 {
1372 bool crcEnabled = protocolOptions & (1 << MOTORON_PROTOCOL_OPTION_CRC_FOR_RESPONSES);
1373 uint8_t byteCount = bus->requestFrom(address, (uint8_t)(length + crcEnabled));
1374 if (byteCount != length + crcEnabled)
1375 {
1376 memset(response, 0, length);
1377 lastError = 50;
1378 return;
1379 }
1380 lastError = 0;
1381 uint8_t * ptr = response;
1382 for (uint8_t i = 0; i < length; i++)
1383 {
1384 *ptr = bus->read();
1385 ptr++;
1386 }
1387 if (crcEnabled && bus->read() != calculateCrc(length, response))
1388 {
1389 lastError = 51;
1390 return;
1391 }
1392 }
1393
1394 void sendCommandAndReadResponse(uint8_t cmdLength, const uint8_t * cmd,
1395 uint8_t responseLength, uint8_t * response)
1396 {
1397 sendCommand(cmdLength, cmd);
1398 if (getLastError())
1399 {
1400 memset(response, 0, responseLength);
1401 return;
1402 }
1403 readResponse(responseLength, response);
1404 }

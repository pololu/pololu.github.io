77 inline uint16_t readBatteryMillivolts()
78 {
79  const uint8_t sampleCount = 8;
80  uint16_t sum = 0;
81  for (uint8_t i = 0; i < sampleCount; i++)
82  {
83  sum += analogRead(A1);
84  }
85 
86  // VBAT = 3 * millivolt reading = 3 * raw * 5000/1024
87  // = raw * 1875 / 128
88  // The correction term below makes it so that we round to the
89  // nearest whole number instead of always rounding down.
90  const uint32_t correction = 64 * sampleCount - 1;
91  return ((uint32_t)sum * 1875 + correction) / (128 * sampleCount);
92 }
